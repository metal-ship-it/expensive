async function updateSilverData() { // ⚠️ 請換上你的 GoldAPI Key const apiKey = "[YOUR_GOLDAPI_KEY]"; try { // 抓取國際白銀 (XAG) 對 美元 (USD) 的價格 const response = await fetch("https://www.goldapi.io/api/XAG/USD", { headers: { "x-access-token": apiKey, "Content-Type": "application/json" } }); const data = await response.json(); const spotPriceUsd = data.price; // 取得國際盎司美元價 const premiumUsd = 10; // 你要求的 +10 美元 const exchangeRate = 32; // 你要求的匯率乘數 32 const weightFactor = 32; // 你要求的重量/換算乘數 32 // 公式：(國際價 + 10) * 32 * 32 const finalPriceKg = (spotPriceUsd + premiumUsd) * exchangeRate * weightFactor; // 回收價邏輯 (假設回收價比賣出價少 2000 元，你可以自行調整) const buyPriceKg = finalPriceKg - 2000; document.getElementById('spot-price').innerText = `$${spotPriceUsd.toFixed(2)} USD/oz`; document.getElementById('sell-price-kg').innerText = `NT$ ${Math.round(finalPriceKg).toLocaleString()}`; document.getElementById('buy-price-kg').innerText = `NT$ ${Math.round(buyPriceKg).toLocaleString()}`; } catch (e) { console.error("報價讀取失敗", e); document.getElementById('sell-price-kg').innerText = "連線錯誤"; } }
